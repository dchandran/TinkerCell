FILE(GLOB SBWCORE_SOURCE *.cpp)

ADD_LIBRARY( SBWCore
  SHARED
  ${SBWCORE_SOURCE}
)

IF (WIN32)
	IF (MINGW)
		TARGET_LINK_LIBRARIES( SBWCore
			SBWCore
			ws2_32
		)
	ELSE(MINGW)
		TARGET_LINK_LIBRARIES( SBWCore
			SBWCore
			wsock32
		)
	ENDIF(MINGW)
ENDIF(WIN32)


 INSTALL(TARGETS SBWCore
	RUNTIME DESTINATION bin/../
	LIBRARY DESTINATION bin/../
	ARCHIVE DESTINATION bin/../
  )
